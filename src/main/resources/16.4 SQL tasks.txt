SELECT I.SUMMARY, I2.NAME
FROM ISSUES I
         INNER JOIN ISSUESLISTS I2 on I.ISSUESLISTS_ID = I2.ID;


SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
         INNER JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID;


SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*)
FROM USERS U
         INNER JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME,U.LASTNAME;
